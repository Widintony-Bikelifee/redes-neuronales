# redes-neuronales
# 
# Proyecto de clasificación utilizando Perceptrón y Adaline con interfaz web en Flask.
# 
# Instalación y ejecución:
# 
# 1. Clonar el repositorio:
#    git clone https://github.com/Widintony-Bikelifee/redes-neuronales
#    cd redes-neuronales
# 
# 2. Crear entorno virtual:
#    python -m venv env
# 
# 3. Activar entorno virtual (Windows PowerShell):
#    Set-ExecutionPolicy -Scope CurrentUser -ExecutionPolicy RemoteSigned
#    .\env\Scripts\activate
# 
# 4. Instalar dependencias:
#    pip install numpy
#    pip install flask
# 
# 5. Ejecutar aplicación:
#    python app.py
# 
# Requisitos:
# 
# - Python 3.13
# - NumPy
# - Flask
